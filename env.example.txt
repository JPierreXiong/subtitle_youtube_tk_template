# ============================================
# Subtitle TK - 环境变量配置示例文件
# ============================================
# 使用方法：
# 1. 复制此文件为 .env.local (开发环境) 或 .env.production (生产环境)
# 2. 填写所有必需的环境变量值
# 3. 根据实际使用的服务，配置相应的可选变量
# ============================================

# ============================================
# 基础应用配置 (必需)
# ============================================
# 应用 URL - 必需
NEXT_PUBLIC_APP_URL=http://localhost:3000

# 应用名称 - 可选，默认: Subtitle TK
NEXT_PUBLIC_APP_NAME=Subtitle TK

# 默认语言 - 可选，默认: en (可选值: en, zh, fr)
NEXT_PUBLIC_DEFAULT_LOCALE=en

# 主题 - 可选，默认: default
NEXT_PUBLIC_THEME=default

# 外观模式 - 可选，默认: system (可选值: system, light, dark)
NEXT_PUBLIC_APPEARANCE=system

# ============================================
# 数据库配置 (必需)
# ============================================
# PostgreSQL 数据库连接 URL - 必需
# 格式: postgresql://用户名:密码@主机:端口/数据库名
DATABASE_URL=postgresql://user:password@localhost:5432/subtitle_tk

# 数据库提供商 - 可选，默认: postgresql (可选值: postgresql, mysql, sqlite)
DATABASE_PROVIDER=postgresql

# ============================================
# 认证配置 (必需)
# ============================================
# 认证密钥 - 必需
# 生成命令: openssl rand -base64 32
AUTH_SECRET=your-auth-secret-key-here-generate-with-openssl-rand-base64-32

# 认证 URL - 可选，默认使用 NEXT_PUBLIC_APP_URL
AUTH_URL=http://localhost:3000

# Google OAuth - 可选
GOOGLE_CLIENT_ID=your-google-client-id
GOOGLE_CLIENT_SECRET=your-google-client-secret

# ============================================
# RapidAPI 配置 (必需 - 用于视频和字幕提取)
# ============================================
# RapidAPI API Key - 必需
# 获取地址: https://rapidapi.com/
NEXT_PUBLIC_RAPIDAPI_KEY=your-rapidapi-key-here

# TikTok 视频下载 API Host - 可选，有默认值
NEXT_PUBLIC_RAPIDAPI_HOST_TIKTOK_DOWNLOAD=tiktok-video-no-watermark2.p.rapidapi.com

# TikTok 字幕提取 API Host - 可选，有默认值
NEXT_PUBLIC_RAPIDAPI_HOST_TIKTOK_TRANSCRIPT=tiktok-transcriptor-api3.p.rapidapi.com

# YouTube 字幕提取 API Host - 可选，有默认值
NEXT_PUBLIC_RAPIDAPI_HOST_YOUTUBE_TRANSCRIPT=youtube-transcripts-transcribe-youtube-video-to-text.p.rapidapi.com

# YouTube 视频下载 API Host - 可选，有默认值
NEXT_PUBLIC_RAPIDAPI_HOST_YOUTUBE_DOWNLOAD=youtube-video-and-shorts-downloader1.p.rapidapi.com

# ============================================
# Gemini AI 配置 (必需 - 用于字幕翻译)
# ============================================
# Google Gemini API Key - 必需
# 获取地址: https://makersuite.google.com/app/apikey
GEMINI_API_KEY=your-gemini-api-key-here

# Gemini Base URL - 可选，有默认值
GEMINI_BASE_URL=https://generativelanguage.googleapis.com/v1beta

# ============================================
# 存储配置 (选择一个存储提供商)
# ============================================

# --- Vercel Blob 存储 (推荐) ---
# Vercel Blob Read/Write Token - 如果使用 Vercel Blob
# 获取地址: Vercel Dashboard -> Storage -> Blob -> Settings
BLOB_READ_WRITE_TOKEN=vercel_blob_rw_your-token-here

# 存储提供商选择 - 可选，默认: vercel-blob (可选值: vercel-blob, r2, s3)
STORAGE_PROVIDER=vercel-blob

# --- Cloudflare R2 存储 (可选) ---
# R2 Account ID - 如果使用 R2
R2_ACCOUNT_ID=your-r2-account-id

# R2 Access Key ID - 如果使用 R2
R2_ACCESS_KEY_ID=your-r2-access-key-id

# R2 Secret Access Key - 如果使用 R2
R2_SECRET_ACCESS_KEY=your-r2-secret-access-key

# R2 Bucket Name - 如果使用 R2
R2_BUCKET_NAME=your-r2-bucket-name

# R2 Endpoint - 可选
R2_ENDPOINT=https://your-account-id.r2.cloudflarestorage.com

# R2 Public Domain - 可选，自定义域名
R2_PUBLIC_DOMAIN=https://your-custom-domain.com

# R2 Region - 可选，默认: auto
R2_REGION=auto

# --- AWS S3 存储 (可选) ---
# S3 Endpoint - 如果使用 S3
S3_ENDPOINT=https://s3.amazonaws.com

# S3 Region - 如果使用 S3
S3_REGION=us-east-1

# S3 Access Key ID - 如果使用 S3
S3_ACCESS_KEY_ID=your-s3-access-key-id

# S3 Secret Access Key - 如果使用 S3
S3_SECRET_ACCESS_KEY=your-s3-secret-access-key

# S3 Bucket Name - 如果使用 S3
S3_BUCKET_NAME=your-s3-bucket-name

# S3 Public Domain - 可选，自定义域名
S3_PUBLIC_DOMAIN=https://your-custom-domain.com

# ============================================
# 邮件配置 (可选)
# ============================================
# Resend API Key - 如果使用 Resend 发送邮件
RESEND_API_KEY=re_your-resend-api-key

# Resend Default From - 可选
RESEND_DEFAULT_FROM=noreply@example.com

# ============================================
# AI 服务配置 (可选 - 用于 AI Chat 功能)
# ============================================
# OpenRouter API Key - 如果使用 OpenRouter
OPENROUTER_API_KEY=sk-or-v1-your-openrouter-key

# Replicate API Key - 如果使用 Replicate
REPLICATE_API_KEY=r8_your-replicate-key

# Fal API Key - 如果使用 Fal
FAL_API_KEY=your-fal-api-key

# Kie API Key - 如果使用 Kie
KIE_API_KEY=your-kie-api-key

# ============================================
# 支付配置 (可选)
# ============================================
# Stripe Secret Key - 如果使用 Stripe
STRIPE_SECRET_KEY=sk_test_your-stripe-secret-key

# Stripe Publishable Key - 如果使用 Stripe
NEXT_PUBLIC_STRIPE_PUBLISHABLE_KEY=pk_test_your-stripe-publishable-key

# Stripe Webhook Secret - 如果使用 Stripe Webhook
STRIPE_WEBHOOK_SECRET=whsec_your-stripe-webhook-secret

# ============================================
# 分析统计配置 (可选)
# ============================================
# Google Analytics ID - 如果使用 Google Analytics
NEXT_PUBLIC_GA_ID=G-XXXXXXXXXX

# Google Tag Manager ID - 如果使用 GTM
NEXT_PUBLIC_GTM_ID=GTM-XXXXXXX

# ============================================
# 其他配置 (可选)
# ============================================
# Node 环境 - 可选，默认: development
NODE_ENV=development

# 调试模式 - 可选，默认: false
NEXT_PUBLIC_DEBUG=false

# ============================================
# 重要提示
# ============================================
# 1. 必需的环境变量：
#    - DATABASE_URL (数据库连接)
#    - AUTH_SECRET (认证密钥)
#    - NEXT_PUBLIC_APP_URL (应用 URL)
#    - NEXT_PUBLIC_RAPIDAPI_KEY (RapidAPI 密钥)
#    - GEMINI_API_KEY (Gemini API 密钥)
#
# 2. 存储配置：
#    - 至少配置一个存储提供商 (Vercel Blob / R2 / S3)
#    - 推荐使用 Vercel Blob (最简单)
#
# 3. 安全提示：
#    - ⚠️ 不要将包含真实密钥的 .env.local 文件提交到 Git
#    - ⚠️ 生产环境使用强密码和密钥
#    - ⚠️ 定期轮换 API 密钥
#
# 4. 优先级说明：
#    - 环境变量优先级高于数据库配置
#    - 如果同时配置了环境变量和数据库配置，环境变量会优先使用
# ============================================
